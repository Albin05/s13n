### Q1 Solution

```python
class Car:
    def __init__(self, make, model, year):
        self.make = make
        self.model = model
        self.year = year

car1 = Car("Toyota", "Camry", 2022)
car2 = Car("Honda", "Civic", 2023)

print(f"{car1.year} {car1.make} {car1.model}")  # 2022 Toyota Camry
print(f"{car2.year} {car2.make} {car2.model}")  # 2023 Honda Civic
```
