### Q3 Solution

```python
class User:
    def __init__(self, name, email, role):
        self.name = name
        self.email = email
        self.role = role

    @classmethod
    def from_string(cls, s):
        name, email, role = s.split(",")
        return cls(name.strip(), email.strip(), role.strip())

u = User.from_string("Alice, alice@mail.com, admin")
print(f"{u.name} ({u.role})")  # Alice (admin)
```
