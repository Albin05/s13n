### Q4 Solution

```python
class Team:
    def __init__(self, name, *members):
        self.name = name
        self.members = list(members)

    def add_member(self, name):
        self.members.append(name)

    def __len__(self):
        return len(self.members)

t = Team("Dev", "Alice", "Bob")
t.add_member("Carol")
print(len(t))        # 3
print(t.members)     # ['Alice', 'Bob', 'Carol']
```
