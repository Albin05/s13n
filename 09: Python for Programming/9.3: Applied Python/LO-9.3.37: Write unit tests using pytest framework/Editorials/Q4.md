## Problem 4: Parameterized Tests

```python
import pytest

def grade_score(score):
    if score >= 90:
        return "A"
    elif score >= 80:
        return "B"
    elif score >= 70:
        return "C"
    elif score >= 60:
        return "D"
    else:
        return "F"

# test_grades.py
@pytest.mark.parametrize("score,expected", [
    (95, "A"),
    (90, "A"),
    (89, "B"),
    (80, "B"),
    (79, "C"),
    (70, "C"),
    (69, "D"),
    (60, "D"),
    (59, "F"),
    (0, "F"),
])
def test_grade_score(score, expected):
    assert grade_score(score) == expected
```
